## INotifyPropertyChanged Generator

The classic use case. Generate property change notifications from annotated fields.

### Marker Attribute (PostInitialization)

```csharp
context.RegisterPostInitializationOutput(static ctx => {
    ctx.AddSource("ObservablePropertyAttribute.g.cs", """
        // <auto-generated/>
        namespace MyMvvm;

        [System.AttributeUsage(System.AttributeTargets.Field)]
        internal sealed class ObservablePropertyAttribute : System.Attribute { }
        """);

    ctx.AddSource("INotifyPropertyChangedHelper.g.cs", """
        // <auto-generated/>
        namespace MyMvvm;

        public abstract class ObservableObject : System.ComponentModel.INotifyPropertyChanged {
            public event System.ComponentModel.PropertyChangedEventHandler? PropertyChanged;

            protected void OnPropertyChanged([System.Runtime.CompilerServices.CallerMemberName] string? name = null) {
                PropertyChanged?.Invoke(this, new System.ComponentModel.PropertyChangedEventArgs(name));
            }

            protected bool SetProperty<T>(ref T field, T value, [System.Runtime.CompilerServices.CallerMemberName] string? name = null) {
                if (System.Collections.Generic.EqualityComparer<T>.Default.Equals(field, value)) {
                    return false;
                }
                field = value;
                OnPropertyChanged(name);
                return true;
            }
        }
        """);
});
```

### Model

```csharp
public readonly record struct ObservableFieldModel(
    string FieldName,
    string PropertyName,
    string TypeName,
    string ContainingTypeName,
    string ContainingNamespace,
    bool IsContainingTypePartial
);
```

### Transform

```csharp
var fields = context.SyntaxProvider.ForAttributeWithMetadataName(
    "MyMvvm.ObservablePropertyAttribute",
    predicate: static (node, _) => node is VariableDeclaratorSyntax,
    transform: static (ctx, _) => {
        var fieldSymbol = (IFieldSymbol)ctx.TargetSymbol;
        var fieldName = fieldSymbol.Name;
        var propertyName = fieldName.TrimStart('_');
        propertyName = char.ToUpper(propertyName[0]) + propertyName[1..];

        return new ObservableFieldModel(
            fieldName,
            propertyName,
            fieldSymbol.Type.ToDisplayString(),
            fieldSymbol.ContainingType.Name,
            fieldSymbol.ContainingType.ContainingNamespace.ToDisplayString(),
            fieldSymbol.ContainingType.DeclaringSyntaxReferences
                .Any(r => r.GetSyntax() is TypeDeclarationSyntax { Modifiers: var m }
                    && m.Any(SyntaxKind.PartialKeyword))
        );
    }
);
```

### Output

```csharp
context.RegisterSourceOutput(fields, static (spc, model) => {
    if (!model.IsContainingTypePartial) {
        spc.ReportDiagnostic(/* ... */);
        return;
    }

    spc.AddSource($"{model.ContainingTypeName}.{model.PropertyName}.g.cs", $$"""
        // <auto-generated/>
        namespace {{model.ContainingNamespace}};

        partial class {{model.ContainingTypeName}} {
            public {{model.TypeName}} {{model.PropertyName}} {
                get => {{model.FieldName}};
                set => SetProperty(ref {{model.FieldName}}, value);
            }
        }
        """);
});
```

### Usage (C# 14 with `field` keyword)

The user's consuming code can also leverage C# 14 `field` keyword for custom validation alongside generated properties:

```csharp
using MyMvvm;

namespace MyApp.ViewModels;

public partial class PersonViewModel : ObservableObject {
    [ObservableProperty]
    private string _firstName = "";

    [ObservableProperty]
    private string _lastName = "";

    // C# 14: field keyword for manual property with validation
    public int Age {
        get;
        set => field = value >= 0 ? value : throw new ArgumentOutOfRangeException(nameof(value));
    }
}
```

## DI Service Registration Generator

Auto-register services decorated with `[RegisterService]`.

```csharp
[Generator]
public sealed class ServiceRegistrationGenerator : IIncrementalGenerator {
    public void Initialize(IncrementalGeneratorInitializationContext context) {
        context.RegisterPostInitializationOutput(static ctx => {
            ctx.AddSource("RegisterServiceAttribute.g.cs", """
                // <auto-generated/>
                namespace MyDI;

                [System.AttributeUsage(System.AttributeTargets.Class)]
                internal sealed class RegisterServiceAttribute : System.Attribute {
                    public RegisterServiceAttribute(
                        System.Type? serviceType = null,
                        ServiceLifetime lifetime = ServiceLifetime.Transient) {
                        ServiceType = serviceType;
                        Lifetime = lifetime;
                    }
                    public System.Type? ServiceType { get; }
                    public ServiceLifetime Lifetime { get; }
                }

                internal enum ServiceLifetime { Transient, Scoped, Singleton }
                """);
        });

        var services = context.SyntaxProvider.ForAttributeWithMetadataName(
            "MyDI.RegisterServiceAttribute",
            predicate: static (n, _) => n is ClassDeclarationSyntax,
            transform: static (ctx, _) => {
                var type = (INamedTypeSymbol)ctx.TargetSymbol;
                var attr = ctx.Attributes[0];
                var serviceType = attr.ConstructorArguments.Length > 0
                    && attr.ConstructorArguments[0].Value is INamedTypeSymbol st
                    ? st.ToDisplayString()
                    : type.Interfaces.FirstOrDefault()?.ToDisplayString()
                      ?? type.ToDisplayString();
                var lifetime = attr.ConstructorArguments.Length > 1
                    ? (int)attr.ConstructorArguments[1].Value!
                    : 0;

                return new ServiceModel(
                    type.ToDisplayString(),
                    serviceType,
                    lifetime switch {
                        1 => "Scoped",
                        2 => "Singleton",
                        _ => "Transient"
                    }
                );
            }
        ).Collect();

        context.RegisterSourceOutput(services, static (spc, svcs) => {
            if (svcs.IsDefaultOrEmpty) return;

            var sb = new StringBuilder("""
                // <auto-generated/>
                using Microsoft.Extensions.DependencyInjection;

                namespace MyDI.Generated;

                public static class ServiceCollectionExtensions {
                    public static IServiceCollection AddGeneratedServices(this IServiceCollection services) {

                """);

            foreach (var svc in svcs.OrderBy(s => s.Implementation, StringComparer.Ordinal)) {
                sb.AppendLine($"        services.Add{svc.Lifetime}<{svc.Interface}, {svc.Implementation}>();");
            }

            sb.AppendLine("        return services;");
            sb.AppendLine("    }");
            sb.AppendLine("}");

            spc.AddSource("ServiceCollectionExtensions.g.cs", sb.ToString());
        });
    }
}

public readonly record struct ServiceModel(
    string Implementation,
    string Interface,
    string Lifetime
);
```

## Enum Utilities Generator

Generate `ToStringFast()`, `TryParse`, and display name resolution without reflection.

```csharp
[Generator]
public sealed class EnumExtensionsGenerator : IIncrementalGenerator {
    public void Initialize(IncrementalGeneratorInitializationContext context) {
        context.RegisterPostInitializationOutput(static ctx => {
            ctx.AddSource("EnumExtensionsAttribute.g.cs", """
                // <auto-generated/>
                namespace MyEnums;

                [System.AttributeUsage(System.AttributeTargets.Enum)]
                internal sealed class EnumExtensionsAttribute : System.Attribute { }
                """);
        });

        var enums = context.SyntaxProvider.ForAttributeWithMetadataName(
            "MyEnums.EnumExtensionsAttribute",
            predicate: static (n, _) => n is EnumDeclarationSyntax,
            transform: static (ctx, _) => {
                var enumSymbol = (INamedTypeSymbol)ctx.TargetSymbol;
                var members = enumSymbol.GetMembers()
                    .OfType<IFieldSymbol>()
                    .Where(f => f.HasConstantValue)
                    .Select(f => new EnumMemberModel(
                        f.Name,
                        f.ConstantValue?.ToString() ?? "0"))
                    .ToImmutableArray();

                return new EnumModel(
                    enumSymbol.Name,
                    enumSymbol.ContainingNamespace.ToDisplayString(),
                    enumSymbol.EnumUnderlyingType?.ToDisplayString() ?? "int",
                    new EquatableArray<EnumMemberModel>(members)
                );
            }
        );

        context.RegisterSourceOutput(enums, static (spc, model) => {
            var sb = new StringBuilder($$"""
                // <auto-generated/>
                namespace {{model.Namespace}};

                public static class {{model.Name}}Extensions {
                    public static string ToStringFast(this {{model.Name}} value) => value switch {

                """);

            foreach (var member in model.Members) {
                sb.AppendLine($"        {model.Name}.{member.Name} => nameof({model.Name}.{member.Name}),");
            }

            sb.AppendLine("""
                        _ => value.ToString()
                    };

                    public static bool TryParseFast(ReadOnlySpan<char> value, out """);
            sb.Append(model.Name);
            sb.AppendLine(""" result) {""");

            foreach (var member in model.Members) {
                sb.AppendLine($"""
                        if (value.SequenceEqual("{member.Name}")) {{
                            result = {model.Name}.{member.Name};
                            return true;
                        }}
                """);
            }

            sb.AppendLine($$"""
                        result = default;
                        return false;
                    }

                    public static int MemberCount => {{model.Members.Length}};
                }
                """);

            spc.AddSource($"{model.Name}Extensions.g.cs", sb.ToString());
        });
    }
}

public readonly record struct EnumModel(
    string Name,
    string Namespace,
    string UnderlyingType,
    EquatableArray<EnumMemberModel> Members
);

public readonly record struct EnumMemberModel(
    string Name,
    string Value
) : IEquatable<EnumMemberModel>;
```

## Strongly-Typed Configuration Generator

Read `.json` or `.yaml` config files as additional files and generate typed classes.

```csharp
// Pipeline: AdditionalFiles → parse JSON → generate classes
var configFiles = context.AdditionalTextsProvider
    .Where(static f => f.Path.EndsWith(".myconfig.json", StringComparison.OrdinalIgnoreCase))
    .Select(static (file, ct) => {
        var text = file.GetText(ct)?.ToString() ?? "{}";
        var name = Path.GetFileNameWithoutExtension(
            Path.GetFileNameWithoutExtension(file.Path)); // Remove .myconfig.json

        // Parse JSON structure (simplified - use System.Text.Json in real code)
        return new ConfigModel(name, ParseProperties(text));
    });

context.RegisterSourceOutput(configFiles, static (spc, model) => {
    var sb = new StringBuilder($$"""
        // <auto-generated/>
        namespace Configuration;

        public sealed class {{model.Name}}Config {
        """);

    foreach (var prop in model.Properties) {
        sb.AppendLine($"    public required {prop.TypeName} {prop.Name} {{ get; init; }}");
    }

    sb.AppendLine("}");
    spc.AddSource($"{model.Name}Config.g.cs", sb.ToString());
});
```

## Builder Pattern Generator

Generate fluent builders for annotated classes:

```csharp
// For a class like:
[GenerateBuilder]
public partial class Order {
    public required string CustomerId { get; init; }
    public required decimal Total { get; init; }
    public string? Notes { get; init; }
}

// Generator produces:
public partial class Order {
    public static OrderBuilder CreateBuilder() => new();

    public sealed class OrderBuilder {
        private string? _customerId;
        private decimal _total;
        private string? _notes;

        public OrderBuilder WithCustomerId(string value) {
            _customerId = value;
            return this;
        }

        public OrderBuilder WithTotal(decimal value) {
            _total = value;
            return this;
        }

        public OrderBuilder WithNotes(string? value) {
            _notes = value;
            return this;
        }

        public Order Build() => new() {
            CustomerId = _customerId ?? throw new InvalidOperationException("CustomerId is required"),
            Total = _total,
            Notes = _notes
        };
    }
}
```

## Mapping Generator

Generate type-to-type mapping code (like AutoMapper but compile-time):

```csharp
// User writes:
[MapTo(typeof(OrderDto))]
public partial class Order {
    public int Id { get; set; }
    public string CustomerName { get; set; } = "";
    public decimal Total { get; set; }
}

public class OrderDto {
    public int Id { get; set; }
    public string CustomerName { get; set; } = "";
    public decimal Total { get; set; }
}

// Generator produces:
public partial class Order {
    public OrderDto ToOrderDto() => new() {
        Id = this.Id,
        CustomerName = this.CustomerName,
        Total = this.Total,
    };
}
```

The generator matches properties by name and type, reports diagnostics for missing mappings.

## Extension Members with field Keyword (C# 14)

C# 14 introduces extension blocks and the `field` keyword. Source generators can emit these modern patterns:

### Generated Extension Properties

```csharp
// Generator can emit extension blocks (C# 14):
extension(IEnumerable<Order> orders) {
    public decimal TotalRevenue => orders.Sum(o => o.Total);
    public int ActiveCount => orders.Count(o => o.IsActive);
}
```

### Generated Properties Using field Keyword

```csharp
// Generator can emit properties with field keyword (C# 14):
public partial class ViewModel {
    // Generated: uses field keyword for lazy initialization
    public string DisplayName => field ??= $"{FirstName} {LastName}";

    // Generated: uses field keyword for change notification
    public string Status {
        get;
        set {
            if (field == value) return;
            field = value;
            OnPropertyChanged();
        }
    }
}
```

### Partial Constructors (C# 14)

C# 14 adds partial constructors, enabling generators to split construction:

```csharp
// User writes (defining declaration):
public partial class MyService {
    public partial MyService(ILogger logger);
}

// Generator emits (implementing declaration):
public partial class MyService {
    private readonly ILogger _logger;

    public partial MyService(ILogger logger) {
        _logger = logger;
        // Generated initialization code
        InitializeGeneratedFields();
    }

    private void InitializeGeneratedFields() {
        // Set up generated state
    }
}
```
